CREATE TABLE IF NOT EXISTS `Promotions` (
  `PromotionID` INT(11) NOT NULL AUTO_INCREMENT,
  `url` text NOT NULL,
  `current_name` text,
  `current_abbrev` text,
  `current_status` text,
  `location` text,
  `active_time` text,
  `www` text,
  `all_names` text,
  `all_abbrev` text,
  `owners` text,
  `popular_events` text,
  `tv_shows` text,
  PRIMARY KEY (`PromotionID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 AUTO_INCREMENT=1 ;
